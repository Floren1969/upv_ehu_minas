def fibo(n):
    try:
        assert(n>0)
        assert(n<20)
    except AssertionError:
        raise ValueError("valores entre 0 y 20")
    a=0
    b=1
    salida=[]
    for x in range(n):
        salida.append(b)
        a,b=b,a+b
    return salida
    
def main():
    """docstring o documentación de la función"""
    try:
        num=input('sartu un numero majeta \n')
        num=int(num)
        print(fibo(num))
    except ValueError:
        print('has metido un valor incorrecto y debe de estar entre 0 y 20')
